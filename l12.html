from transformers import pipeline generator = pipeline("text-generation", model="gpt2") prompt = "Once there was a dragon" result = generator(prompt, max_length=50, num_return_sequences=1) print(result[0]['generated_text']) prompts = ["Tell a story about a robot.", "Describe a futuristic city.", "Write a poem about love."] for prompt in prompts: result = generator(prompt, max_length=50, num_return_sequences=1) print(result[0]['generated_text'])
